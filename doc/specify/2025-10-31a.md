This spec is to implement "Strategy 4: Lightweight Continuous Monitoring" from
[explore/2025-10-30b.md](../explore/2025-10-30b.md). See also
[explore/2025-10-23b.md](../explore/2025-10-23b.md).

For ongoing performance tracking (not detailed profiling) during training in production:

```python
import time
import torch.cuda

class PerformanceMonitor:
    def __init__(self):
        self.timings = {'data_load': [], 'forward': [], 'backward': [], 'optimizer': []}

    def train_epoch(self, model, dataloader, criterion, optimizer):
        model.train()
        data_load_start = time.time()

        for data, target in dataloader:
            self.timings['data_load'].append(time.time() - data_load_start)

            # Forward
            forward_start = time.time()
            output = model(data)
            loss = criterion(output, target)
            torch.cuda.synchronize()  # Wait for GPU
            self.timings['forward'].append(time.time() - forward_start)

            # Backward
            backward_start = time.time()
            loss.backward()
            torch.cuda.synchronize()
            self.timings['backward'].append(time.time() - backward_start)

            # Optimizer
            opt_start = time.time()
            optimizer.step()
            optimizer.zero_grad()
            torch.cuda.synchronize()
            self.timings['optimizer'].append(time.time() - opt_start)

            data_load_start = time.time()

    def report(self):
        import numpy as np
        for key, times in self.timings.items():
            print(f"{key}: {np.mean(times)*1000:.2f}ms (±{np.std(times)*1000:.2f}ms)")

monitor = PerformanceMonitor()
monitor.train_epoch(model, dataloader, criterion, optimizer)
monitor.report()
```
