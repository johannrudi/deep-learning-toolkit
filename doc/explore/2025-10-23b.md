A staged plan works best:

1. micro-time each training-step phase;
2. capture CPU+CUDA timelines;
3. dive to GPU-kernel metrics;
4. isolate the dataloader;
5. inspect torch.compile graph breaks;
6. check memory/throughput.

Use the snippets/commands below; run them in isolation to localize the bottleneck, then
change one factor at a time (batch size, AMP, compile, workers) and re-measure.  Start
with micro-timers and the PyTorch Profiler; escalate to Nsight Systems/Compute and CPU
profilers only if the hotspot is unclear.  If you share your train loop entrypoint, I

# 1) insertable micro-timers: measure data wait, H2D, fwd, bwd, opt, total
import time, torch

def step_with_timing(model, optimizer, loss_fn, batch, device="cuda"):
    # measure data wait (time spent before batch is available)
    t0 = time.perf_counter()  # call this right after previous optimizer.step()
    xb, yb = batch
    t_data = time.perf_counter() - t0

    # record GPU phase times with CUDA events
    e = {k: (torch.cuda.Event(True), torch.cuda.Event(True)) for k in ["h2d","fwd","bwd","opt"]}
    torch.cuda.synchronize()

    e["h2d"][0].record()
    xb = xb.to(device, non_blocking=True); yb = yb.to(device, non_blocking=True)
    e["h2d"][1].record()

    optimizer.zero_grad(set_to_none=True)
    e["fwd"][0].record()
    logits = model(xb)
    loss = loss_fn(logits, yb)
    e["fwd"][1].record()

    e["bwd"][0].record()
    loss.backward()
    e["bwd"][1].record()

    e["opt"][0].record()
    optimizer.step()
    e["opt"][1].record()

    torch.cuda.synchronize()
    ms = {k: e[k][0].elapsed_time(e[k][1]) for k in e}
    ms["total"] = sum(ms.values())
    return loss.item(), ms, t_data

# 5) dataloader sweep: find best num_workers/prefetch/pin_memory; measure data wait and throughput
from torch.utils.data import DataLoader
import itertools, time, torch

def bench_loader(dataset, device="cuda", batch_size=256, trials=100, warmup=20):
    def measure(loader):
        it = iter(loader); t_data = 0.0
        # warm up
        for _ in range(warmup):
            xb, yb = next(it); xb = xb.to(device, non_blocking=True); _ = yb.to(device, non_blocking=True)
        torch.cuda.synchronize()
        t0 = time.perf_counter()
        for _ in range(trials):
            t_start = time.perf_counter()
            xb, yb = next(it)
            t_data += time.perf_counter() - t_start
            xb = xb.to(device, non_blocking=True); _ = yb.to(device, non_blocking=True)
        torch.cuda.synchronize()
        t_tot = time.perf_counter() - t0
        return {"data_ms": 1e3*t_data/trials, "step_ms": 1e3*t_tot/trials, "ips": batch_size/(t_tot/trials)}
    configs = []
    for nw, pf, pin in itertools.product([0,2,4,8], [2,4], [False, True]):
        loader = DataLoader(dataset, batch_size=batch_size, shuffle=True, num_workers=nw,
                            prefetch_factor=pf if nw>0 else None, persistent_workers=bool(nw),
                            pin_memory=pin, pin_memory_device='cuda' if pin else '')
        m = measure(loader); m.update(dict(nw=nw, pf=pf, pin=pin)); configs.append(m)
    return sorted(configs, key=lambda d: (-d["ips"], d["data_ms"]))
